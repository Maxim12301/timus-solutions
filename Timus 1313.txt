n = int(input())

lines = []
x, y = 0, 0
ans = ''


for i in range(n):
    lines.append(list(input().split()))
    
while y != n:
    for i in range(y, -1, -1):
        ans += lines[i][x] + ' '
        x += 1
    x = 0
    y += 1
    

min_x = 1
min_y = 0
y = n - 1
x = min_x

while min_y != n - 1:
    
    for i in range(y, min_y, -1):
        ans += lines[i][x] + ' '
        x += 1
    
    min_y += 1
    min_x += 1
    x = min_x
    
print(ans)
